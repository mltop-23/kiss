вот таблицы
CREATE TABLE Users (
    ID SERIAL PRIMARY KEY,
    FamilyID INT,
    Username VARCHAR(255) NOT NULL,
    Password VARCHAR(255) NOT NULL,
    Email VARCHAR(255) NOT NULL,
    FirstName VARCHAR(255) NOT NULL,
    LastName VARCHAR(255) NOT NULL,
    Gender VARCHAR(50) NOT NULL,
    Role VARCHAR(50) NOT NULL
);

CREATE TABLE Families (
    ID SERIAL PRIMARY KEY,
    HusbandID INT NOT NULL,
    WifeID INT NOT NULL,
    Kisses INT NOT NULL,
    Debt INT NOT NULL,
    CONSTRAINT fk_husband FOREIGN KEY (HusbandID) REFERENCES Users(ID),
    CONSTRAINT fk_wife FOREIGN KEY (WifeID) REFERENCES Users(ID)
);

CREATE TABLE Dishes (
    ID SERIAL PRIMARY KEY,
    Name VARCHAR(255) NOT NULL,
    Recipe TEXT NOT NULL,
    CookingTime INT NOT NULL,
    Complexity VARCHAR(50) NOT NULL,
    Taste VARCHAR(50) NOT NULL,
    Kisses INT NOT NULL
);

CREATE TABLE Meals (
    ID SERIAL PRIMARY KEY,
    Name VARCHAR(255) NOT NULL
);

CREATE TABLE Orders (
    ID SERIAL PRIMARY KEY,
    DishID INT NOT NULL,
    FamilyID INT NOT NULL,
    Status VARCHAR(50) NOT NULL,
    KissesPaid INT NOT NULL,
    CONSTRAINT fk_dish FOREIGN KEY (DishID) REFERENCES Dishes(ID),
    CONSTRAINT fk_family_order FOREIGN KEY (FamilyID) REFERENCES Families(ID)
);

-- Updating the Users table to set FamilyID as a foreign key after Families table is created
ALTER TABLE Users
ADD CONSTRAINT fk_family FOREIGN KEY (FamilyID) REFERENCES Families(ID);




удаление таблиц

-- Удаление внешнего ключа из таблицы Users
ALTER TABLE Users DROP CONSTRAINT IF EXISTS fk_family;

-- Удаление таблиц
DROP TABLE IF EXISTS Orders;
DROP TABLE IF EXISTS Meals;
DROP TABLE IF EXISTS Dishes;
DROP TABLE IF EXISTS Families;
DROP TABLE IF EXISTS Users;
